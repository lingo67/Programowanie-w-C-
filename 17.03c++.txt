#include <string>
#include <iostream>
#include <random>

namespace strutil {
    std::string convertCase(const std::string &s, bool upperCase) {
        std::string s2=s;
        if (upperCase==1) {
            for (int i=0; i<s.length();i++) {
                s2[i]=std::toupper(s2[i]);
            }
        }
        else {
            for (int i=0; i<s.length();i++) {
                s2[i]=std::tolower(s2[i]);
            }
        }
        return s2;
    }

    bool isPalindrome(const std::string &s) {
        std::string s2="";
        for (int i=s.length()-1;i>=0;i--) {
            s2+=s[i];
        }
        if (s==s2) {
            return 1;
        }
        else {
            return 0;
        }
    }

    std::string rand(size_t n) {
        std::string s="";
        for (int i=0;i<n;i++) {
            s+='a'+::rand()%('z'-'a'+1);
        }
        return s;
    }

    size_t longestWordLength(const std::string &s) {
        std::string napis = "";
        std::string max = "";
        for (size_t i = 0; i<=s.length(); i++) {
            if (i<s.length() && s[i]!=' ') {
                napis += s[i];
            }
            else {
                if (napis.length() >= max.length()) {
                    max = napis;
                }
                napis="";
            }
        }
        return max.length();
    }

    void rot13(std::string &s) {
        for (char&c : s) {
            if (c>='a' && c<='z') {
                c='a' + (c-'a'+13)%26;
            }
            else if (c>='A' && c<= 'Z') {
                c='A' + (c-'A'+13)%26;
            }
        }
    }

    size_t countOccurence(const std::string &s1, const std::string &s2) {
        int ile = 0;
        int znak = 0;
        for (int i=0; i<s1.length(); i++) {
            if (s1[i]==s2[0]) {
                znak=0;
                for (int j=0; j<=s2.length()-1;j++) {
                    if (s1[i+j]==s2[j]) {
                        znak+=1;
                    }
                }
                if (znak==s2.length()) {
                    ile+=1;
                }
            }
        }
        return ile;
    }

}

int main() {
    std::cout<<"covertCase: "<<strutil::convertCase("abCDeFghI", 0)<<std::endl;
    std::cout<<"isPalindrome: "<<strutil::isPalindrome("kajak")<<std::endl;
    std::cout<<"rand: "<<strutil::rand(5)<<std::endl;
    std::cout<<"longestWordLength: "<<strutil::longestWordLength("ab bc abcd bcdefgh acd")<<std::endl;
    std::string text = "Hello";
    strutil::rot13(text);
    std::cout<<"rot13: "<<text<<std::endl;
    std::cout<<"countOccurence: "<<strutil::countOccurence("abcOOabcOabc", "abc")<<std::endl;
}
